R1 EQU 0x01
R2 EQU 0x02
R3 EQU 0x03
 
 ORG 00H
 
 ; CLEARING ALL THE TRISTATE BUFFERS AND OUPUTS
 CLRF TRISA 
 CLRF TRISB 
 CLRF TRISC
 CLRF TRISD 
 CLRF TRISE
 
 ;CLEARING TO SEND 0
 CLRF PORTA
 CLRF PORTB
 CLRF PORTC
 CLRF PORTD
 CLRF PORTE
 
 AGAIN: 
    MOVLW B'00000001' 
    MOVWF PORTA 
    CALL DELAY ;DELAY OF 0.4MICRO SECOND
    MOVLW B'00000010' 
    MOVWF PORTA 
    CALL DELAY 
    MOVLW B'00000100' 
    MOVWF PORTA 
    CALL DELAY 
    MOVLW B'00001000' 
    MOVWF PORTA 
    CALL DELAY 
    MOVLW B'00010000' 
    MOVWF PORTA 
    CALL DELAY 
    MOVLW B'00100000' 
    MOVWF PORTA 
    CALL DELAY 
    MOVLW B'01000000' 
    MOVWF PORTA 
    CALL DELAY 
    MOVLW B'10000000' 
    MOVWF PORTA 
    CALL DELAY 
    CLRF PORTA ; SHIFTING 8 TIMES AND CLEARING
    
    ;DO THIS FOR PORT B, C, D (COPY PASTE)
    
    
    MOVLW B'001'
    MOVWF PORTE 
    CALL DELAY 
    MOVLW B'010' 
    MOVWF PORTE 
    CALL DELAY 
    MOVLW B'100' 
    MOVWF PORTE 
    CALL DELAY 
    CLRF PORTE
 GOTO AGAIN 
    
 DELAY: ; TIME PERIOD OF APPROXIMATE 1 SECOND
    MOVLW D'8' 
    MOVWF R3 
    AGAIN4: MOVLW D'48' 
    MOVWF R2 
    AGAIN5: MOVLW D'125' 
    MOVWF R1
    AGAIN6: DECFSZ R1 
    GOTO AGAIN6 
    DECFSZ R2 
    GOTO AGAIN5
    DECFSZ R3 
    GOTO AGAIN4 
    RETURN 


END
